    values_to = "Change"

  mutate(
    Type = case_when(
      str_detect(Period, "households") ~ "Households",
      str_detect(Period, "housing units") ~ "Housing units",
      TRUE ~ NA_character_
    )
  ) %>%
  mutate(PeriodLabel = case_when(
